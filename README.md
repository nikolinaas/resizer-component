## Install

```bash
npm install --save resizer-component
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'resizer-component'
import 'resizer-component/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
